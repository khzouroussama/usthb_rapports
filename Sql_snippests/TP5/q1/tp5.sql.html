CREATE OR REPLACE TRIGGER MIS_A_JOUR_CLIENT
AFTER INSERT OR UPDATE OR DELETE ON CLIENT
FOR EACH ROW

 BEGIN 

   IF INSERTING THEN
   DBMS_output.put_line ('un nouvel client a été ajouté');
     END IF;
  IF UPDATING THEN
   DBMS_output.put_line ('un client a été mis à jour');
     END IF;
  IF DELETING THEN
   DBMS_output.put_line ('le client a été supprimé');
     END IF;
END;
