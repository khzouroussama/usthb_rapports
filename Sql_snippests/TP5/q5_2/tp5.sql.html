DECLARE
  CURSOR EMP IS
  SELECT NUMEMPLOYE from INTERVENANTS;
  NUM EMPLOYE.NUMEMPLOYE%type ;

BEGIN
	OPEN EMP ;
	LOOP
	  FETCH EMP INTO NUM ;
	  EXIT WHEN EMP%notfound;

	  UPDATE EMPLOYE e 
	  SET TOTAL_INTERVENTIONS = TOTAL_INTERVENTIONS + 1 
	  where e.NUMEMPLOYE = NUM ;
	end loop;
	CLOSE EMP ;
end;
/
