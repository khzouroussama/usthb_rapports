CREATE TABLE Employe(
  numEmploye       INTEGER,
  nomEmploye       VARCHAR2(100),
  prenomEmploye    VARCHAR2(100),
  categorieEmploye VARCHAR2(100),
  salairEploye     FLOAT,
  CONSTRAINT PK_Employe PRIMARY KEY (numEmploye),
  CONSTRAINT CK_categorieEmploye
    CHECK (categorieEmploye in ('MECANICIEN’, ’ASSISTANT'))
);
