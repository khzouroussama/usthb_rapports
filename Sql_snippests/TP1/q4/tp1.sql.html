CREATE TABLE Client(
  numClient     INTEGER,
  Civ           VARCHAR2(5),
  prenomClient  VARCHAR2(100),
  nomClient     VARCHAR2(100),
  dateNaissance DATE,
  adresse       VARCHAR2(200),
  telProf       VARCHAR2(100),
  telPriv       VARCHAR2(100),
  fax           VARCHAR2(100),
  CONSTRAINT PK_Client PRIMARY KEY (numClient),
  CONSTRAINT CK_Civ CHECK (Civ in ('M', 'Mle', 'Mme'))
);
